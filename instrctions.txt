Project Setup
        => git & node installation
            sudo apt update && sudo apt upgrade -y

            sudo apt install -y curl git

            curl -fsSL https://deb.nodesource.com/setup_16.x | sudo -E bash -

            sudo apt install -y nodejs build-essential

            # Verify installation
                node -v
                npm -v

            #clone project
               sudo mkdir -p /var/www/html && cd /var/www/html
               sudo git clone https://github.com/engrghulamali/react-devops-aws-azure-gcp.git

            #project permission setup
            # Change ownership to user (recursively for all files)
                sudo chown -R $USER:$USER /var/www/html/react-devops-aws-azure-gcp

            # Set safe permissions for directories and files
                find /var/www/html/react-devops-aws-azure-gcp -type d -exec chmod 755 {} \;
                find /var/www/html/react-devops-aws-azure-gcp -type f -exec chmod 644 {} \;

            #installation
                # Install dependencies (ignore peer deps conflicts for React 17)
                npm install --legacy-peer-deps

            # Build React app
                npm run build

            #nginx setup
                sudo apt install nginx -y

            #default conf file setp
                sudo rm /etc/nginx/sites-enabled/default
                sudo nano /etc/nginx/sites-available/react-site

            #paste this
                server {
                    listen 80;
                    server_name _;

                    root /var/www/html/react-devops-aws-azure-gcp/build;
                    index index.html;

                    location / {
                        try_files $uri /index.html;
                    }
                }

            # Restart nginx
                sudo ln -s /etc/nginx/sites-available/react-site /etc/nginx/sites-enabled/
                sudo nginx -t
                sudo systemctl restart nginx
